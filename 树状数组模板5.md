#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

const int N = 1010;

int n, q;

int arr[N][N];
int tr[N][N];

int lowbit(int x) {
    return x & -x;
}

// 单点修改：加上一个数
void add(int x, int y, int c) {
    for (int i = x; i <= n; i += lowbit(i))
        for (int j = y; j <= n; j += lowbit(j))
            tr[i][j] += c;
}

// 前缀求和
int query(int x, int y) {
    int res = 0;
    for (int i = x; i > 0; i -= lowbit(i))
        for (int j = y; j > 0; j -= lowbit(j))
            res += tr[i][j];
    return res;
}

// 区间求和
int range_query(int x1, int y1, int x2, int y2) {
    return query(x2, y2) - query(x1 - 1, y2) - query(x2, y1 - 1) + query(x1 - 1, y1 - 1);
}

/*
    二维版本的 [单点修改 + 区间求和]
    使用二维树状数组解决
*/
int main() {
    cin >> n >> q;

    for (int i = 1; i <= n; i ++ )
        for (int j = 1; j <= n; j ++ ) {
            scanf("%d", &arr[i][j]);
            add(i, j, arr[i][j]);
        }
    
    int op;
    int x1, y1, x2, y2;
    while (q -- ) {
        scanf("%d", &op);
        if (op == 1) {
            scanf("%d%d", &x1, &y1);
            if (arr[x1][y1] == 0) {
                add(x1, y1, 1);
                arr[x1][y1] = 1;
            } else {
                add(x1, y1, -1);
                arr[x1][y1] = 0;
            }
        } else {
            scanf("%d%d%d%d", &x1, &y1, &x2, &y2);
            printf("%d\n", range_query(x1, y1, x2, y2));
        }
    }

    return 0;
}
