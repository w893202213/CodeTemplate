### 这是洛谷题: P4513
### 在 query 时专门设计了一个结构体作为返回值，值得好好研究

```cpp
#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
#include <vector>

using namespace std;

typedef long long LL;

const int N = 5e5 + 10;

int n, q;

LL arr[N];
// 维护：区间和, 最大左段和, 最大右段和, 最大子段和
LL sum[N << 2], lmax[N << 2], rmax[N << 2], tmax[N << 2];

void up(int u) {
    sum[u] = sum[u << 1] + sum[u << 1 | 1];
    lmax[u] = max(lmax[u << 1], sum[u << 1] + lmax[u << 1 | 1]);
    rmax[u] = max(rmax[u << 1 | 1], sum[u << 1 | 1] + rmax[u << 1]);
    tmax[u] = max({tmax[u << 1], tmax[u << 1 | 1], rmax[u << 1] + lmax[u << 1 | 1]});
}

void build(int l, int r, int u) {
    if (l == r)
        sum[u] = lmax[u] = rmax[u] = tmax[u] = arr[l];
    else {
        int mid = l + r >> 1;
        
        build(l, mid, u << 1);
        build(mid + 1, r, u << 1 | 1);
        
        up(u);
    }
}

void add(int pos, LL jobv, int l, int r, int u) {
    if (l == r)
        sum[u] += jobv, lmax[u] += jobv, rmax[u] += jobv, tmax[u] += jobv;
    else {
        int mid = l + r >> 1;
        
        // 只有单点修改这里是 if-else, 其他都是if-if
        if (pos <= mid)
            add(pos, jobv, l, mid, u << 1);
        else
            add(pos, jobv, mid + 1, r, u << 1 | 1);
        
        up(u);
    }
}

// 求 [jobl, jobr] 的最大子段和, 返回一个四元组
// 表示 {sum, lmax, rmax, tmax}
struct Node {
    LL sum = -2e9;
    LL lmax = -2e9;
    LL rmax = -2e9;
    LL tmax = -2e9;
};

Node query(int jobl, int jobr, int l, int r, int u) {
    if (jobl <= l && r <= jobr)
        return {sum[u], lmax[u], rmax[u], tmax[u]};
    else {
        int mid = l + r >> 1;
        
        if (jobl > mid)  // 只有右边
            return query(jobl, jobr, mid + 1, r, u << 1 | 1);
        
        if (mid + 1 > jobr)  // 只有左边
            return query(jobl, jobr, l, mid, u << 1);
        
        Node left = query(jobl, jobr, l, mid, u << 1);
        Node right = query(jobl, jobr, mid + 1, r, u << 1 | 1);
        Node res;
        
        res.sum = left.sum + right.sum;
        res.lmax = max(left.lmax, left.sum + right.lmax);
        res.rmax = max(right.rmax, right.sum + left.rmax);
        res.tmax = max({left.tmax, right.tmax, left.rmax + right.lmax});
        
        return res;
    }
}

/*
    单点修改, 无需维护懒标记
*/

int main() {
    cin >> n >> q;
    
    for (int i = 1; i <= n; i ++ )
        scanf("%lld", &arr[i]);
    
    build(1, n, 1);
    
    int op, a, b;
    while (q -- ) {
        scanf("%d%d%d", &op, &a, &b);
        if (op == 1) {
            if (a > b)  swap(a, b);
            printf("%lld\n", query(a, b, 1, n, 1).tmax);
        } else {
            add(a, b - arr[a], 1, n, 1);
            arr[a] = b;
        }
    }
    
    return 0;
}
```
