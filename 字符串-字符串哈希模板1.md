#### 最直接的模板题：洛谷 P3370 
#### 给定 n 个字符串，求这些字符串共有多少种 (其实就是unordered_map和unordered_set的功能)

代码：

```cpp
#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
#include <string>

using namespace std;

typedef unsigned long long ULL;

const int N = 1e4 + 10;

ULL nums[N];

ULL get_hash(string s) {
    ULL h = 0;
    const ULL base = 131;
    
    for (char c: s)
        h = h * base + c;
    
    return h;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    
    int n;
    cin >> n;
    
    string s;
    for (int i = 1; i <= n; i ++ ) {
        cin >> s;
        nums[i] = get_hash(s);
    }
    
    sort(nums + 1, nums + n + 1);
    
    int res = 0;
    for (int i = 1; i <= n; i ++ )
        if (i == 1 || nums[i] != nums[i - 1])
            res ++;
    
    cout << res << endl;
    
    return 0;
}
```
