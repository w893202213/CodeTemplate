#include <iostream>

using namespace std;

typedef long long LL;

const int N = 100010;

int n, q;
LL arr[N];
LL tree[N << 2];
LL add_val[N << 2];

void up(int u) {
	tree[u] =  tree[u << 1] + tree[u << 1 | 1];
}

void lazy(int u, LL val, int len) {
	tree[u] += val * len;
	add_val[u] += val;
}

void down(int u, int lcnt, int rcnt) {
	if (add_val[u] != 0) {
		lazy(u << 1, add_val[u], lcnt);
		lazy(u << 1 | 1, add_val[u], rcnt);
		add_val[u] = 0;		
	}
}

void build(int l, int r, int u) {
	if (l == r)
		tree[u] = arr[l];
	else {
		int mid = l + r >> 1;
		build(l, mid, u << 1);
		build(mid + 1, r, u << 1 | 1);
		up(u);
	}
	add_val[u] = 0;
}

LL query(int jobl, int jobr, int l, int r, int u) {
	if (jobl <= l && r <= jobr)
		return tree[u];
	else {
		int mid = l + r >> 1;
		down(u, mid - l + 1, r - mid);
		
		LL res = 0;
		if (jobl <= mid)  res += query(jobl, jobr, l, mid, u << 1);
		if (mid + 1 <= jobr)  res += query(jobl, jobr, mid + 1, r, u << 1 | 1);
		return res;
	}
}

void add(int jobl, int jobr, LL jobv, int l, int r, int u) {
	if (jobl <= l && r <= jobr)
		lazy(u, jobv, r - l + 1);
	else {
		int mid = l + r >> 1;
		down(u, mid - l + 1, r - mid);
		
		if (jobl <= mid)  add(jobl, jobr, jobv, l, mid, u << 1);
		if (mid + 1 <= jobr)  add(jobl, jobr, jobv, mid + 1, r, u << 1 | 1);
		
		up(u);
	}
}

int main() {
	ios::sync_with_stdio(false);
	cin.tie(nullptr);
	
    cin >> n >> q;
    
    for (int i = 1; i <= n; i ++ )
    	cin >> arr[i];
    
    build(1, n, 1);
    
    int op, x, y;
	LL k;
    while (q -- ) {
    	cin >> op >> x >> y;
    	if (op == 1) {
    		cin >> k;
    		add(x, y, k, 1, n, 1);
		} else {
			cout << query(x, y, 1, n, 1) << endl;
		}
	}

    return 0;
}
