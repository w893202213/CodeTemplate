#### 权值树状数组
#### 核心思想：下标 = 数值大小或离散化后的排名
#### query(x) = 当前已经出现的、值 <= x 的数的个数
#### 经典题：洛谷 P1908 逆序对
#### 这是一种经典的方法，可以实现维护"左边有多少个更小的数"、"右边有多少个更大的数"等

代码：

```cpp
#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
#include <string>
#include <vector>

using namespace std;

typedef long long LL;

const int N = 5e5 + 10;

int n;
int nums[N], sorted[N];

LL tr[N];

int lowbit(int x) {
    return x & -x;
}

void add(int x, LL c) {
    for (int i = x; i <= n; i += lowbit(i))
        tr[i] += c;
}

LL query(int x) {
    LL res = 0;
    for (int i = x; i; i -= lowbit(i))
        res += tr[i];
    return res;
}

LL range_query(int l, int r) {
    return query(r) - query(l - 1);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    
    cin >> n;
    
    for (int i = 1; i <= n; i ++ ) {
        cin >> nums[i];
        sorted[i] = nums[i];
    }
    
    // 排序
    sort(sorted + 1, sorted + n + 1);
    
    // 去重
    int l, r;
    for (l = 1, r = 1; r <= n; r ++ ) {
        while (r <= n && sorted[l] == sorted[r])  r ++;
        if (r > n)  break;
        l ++;
        sorted[l] = sorted[r];
    }
    
    int maxl = l;
    
    // 二分覆盖原值
    for (int i = 1; i <= n; i ++ ) {
        int l = 1, r = maxl;
        while (l < r) {
            int mid = l + r >> 1;
            if (sorted[mid] >= nums[i])  r = mid;
            else  l = mid + 1;
        }
        nums[i] = r;
    }
    
    long long res = 0;
    for (int i = n; i >= 1; i -- ) {
        // 加上已有的(后面位置的)比当前值更小的
        res += range_query(1, nums[i] - 1);
        add(nums[i], 1);
    }
    
    cout << res << endl;
    
    return 0;
}
```
