#### 差分树状数组
#### 和标准的树状数组正好对称，差分树状数组维护的操作为: 区间修改 + 单点查询
#### 维护原数组的差分数组的树状数组即可

代码：

```cpp
#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
#include <string>
#include <vector>

using namespace std;

typedef long long LL;

const int N = 5e5 + 10;

int n;
LL a[N], tr[N];

int lowbit(int x) {
    return x & -x;
}

void add(int x, LL c) {
    for (int i = x; i <= n;i += lowbit(i))
        tr[i] += c;
}

LL query(int x) {
    LL res = 0;
    for (int i = x; i; i -= lowbit(i))
        res += tr[i];
    return res;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    
    int q;
    cin >> n >> q;
    
    for (int i = 1; i <= n; i ++ ) {
        cin >> a[i];
        add(i, a[i] - a[i - 1]);
    }
    
    int op, x, y;
    LL c;
    
    while (q -- ) {
        cin >> op;
        if (op == 1) {
            cin >> x >> y >> c;
            add(x, c);
            add(y + 1, -c);
        } else {
            cin >> x;
            cout << query(x) << endl;
        }
    }
    
    return 0;
}
```
