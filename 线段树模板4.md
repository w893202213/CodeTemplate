#### 区间覆盖 + 区间加值 + 区间求最大值

##### 核心是down函数和两个lazy函数的设计： 
##### lazy函数修改具体的数组值，down函数调用lazy函数即可 

代码：

```cpp
#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
#include <cmath>

using namespace std;

typedef long long LL;

const int N = 1e6 + 10;

int n, q;

LL arr[N];
LL maxv[N << 2], add_val[N << 2], set_val[N << 2];
bool has_set[N << 2];

void up(int u) {
    maxv[u] = max(maxv[u << 1], maxv[u << 1 | 1]);
}

void lazy_add(int u, LL val) {
    if (has_set[u]) {
        set_val[u] += val;
    } else {
        add_val[u] += val;
    }
    maxv[u] += val;
}

void lazy_set(int u, LL val) {
    has_set[u] = true;
    set_val[u] = val;
    add_val[u] = 0;
    maxv[u] = val;
}

void down(int u) {
    if (has_set[u]) {
        lazy_set(u << 1, set_val[u]);
        lazy_set(u << 1 | 1, set_val[u]);
        has_set[u] = false;
    } else if (add_val[u]) {
        lazy_add(u << 1, add_val[u]);
        lazy_add(u << 1 | 1, add_val[u]);
        add_val[u] = 0;
    }
}

void build(int l, int r, int u) {
    if (l == r)
        maxv[u] = arr[l];
    else {
        int mid = l + r >> 1;
        
        build(l, mid, u << 1);
        build(mid + 1, r, u << 1 | 1);
        
        up(u);
    }
    
    add_val[u] = 0;
    set_val[u] = 0;
    has_set[u] = false;
}

void set(int jobl, int jobr, LL jobv, int l, int r, int u) {
    if (jobl <= l && r <= jobr)
        lazy_set(u, jobv);
    else {
        int mid = l + r >> 1;
        
        down(u);
        
        if (jobl <= mid)
            set(jobl, jobr, jobv, l, mid, u << 1);
        if (mid + 1 <= jobr)
            set(jobl, jobr, jobv, mid + 1, r, u << 1 | 1);
        
        up(u);
    }
}

void add(int jobl, int jobr, LL jobv, int l, int r, int u) {
    if (jobl <= l && r <= jobr)
        lazy_add(u, jobv);
    else {
        int mid = l + r >> 1;
        
        down(u);
        
        if (jobl <= mid)
            add(jobl, jobr, jobv, l, mid, u << 1);
        if (mid + 1 <= jobr)
            add(jobl, jobr, jobv, mid + 1, r, u << 1 | 1);
        
        up(u);
    }
}

LL query(int jobl, int jobr, int l, int r, int u) {
    if (jobl <= l && r <= jobr)
        return maxv[u];
    else {
        int mid = l + r >> 1;
        
        down(u);
        
        LL res = -1e18;
        if (jobl <= mid)
            res = max(res, query(jobl, jobr, l, mid, u << 1));
        if (mid + 1 <= jobr)
            res = max(res, query(jobl, jobr, mid + 1, r, u << 1 | 1));
        return res;
    }
}

int main() {
    cin >> n >> q;
    
    for (int i = 1; i <= n; i ++ )
        scanf("%lld", &arr[i]);
    
    build(1, n, 1);
    
    int op, l, r;
    LL x;
    while (q -- ) {
        scanf("%d%d%d", &op, &l, &r);
        if (op == 1) {
            scanf("%lld", &x);
            set(l, r, x, 1, n, 1);
        } else if (op == 2) {
            scanf("%lld", &x);
            add(l, r, x, 1, n, 1);
        } else {
            printf("%lld\n", query(l, r, 1, n, 1));
        }
    }
    
    return 0;
}
```
