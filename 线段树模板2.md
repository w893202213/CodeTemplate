#### 线段树模板2：区间修改(覆盖)、区间查询(求最大值)

代码：

```cpp
#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

typedef long long LL;

const int N = 2e5 + 10;

int n, q;
int arr[N];
LL tree[N << 2];

void up(int u) {
    tree[u] = max(tree[u << 1], tree[u << 1 | 1]);
}

void build(int l, int r, int u) {
    if (l == r)
        tree[u] = arr[l];
    else {
        int mid = l + r >> 1;
        
        build(l, mid, u << 1);
        build(mid + 1, r, u << 1 | 1);
        
        up(u);
    }
}

// 将 [jobl, jobr] 的数 "更改为" jobv
void modify(int jobl, int jobr, LL jobv, int l, int r, int u) {
    if (jobl <= l && r <= jobr)
        tree[u] = jobv;
    else {
        int mid = l + r >> 1;
        
        if (jobl <= mid)
            modify(jobl, jobr, jobv, l, mid, u << 1);
        if (mid + 1 <= jobr)
            modify(jobl, jobr, jobv, mid + 1, r, u << 1 | 1);
        
        up(u);
    }
}

LL query(int jobl, int jobr, int l, int r, int u) {
    if (jobl <= l && r <= jobr)
        return tree[u];
    else {
        int mid = l + r >> 1;
        
        LL res = -2e9;
        if (jobl <= mid)
            res = max(res, query(jobl, jobr, l, mid, u << 1));
        if (mid + 1 <= jobr)
            res = max(res, query(jobl, jobr, mid + 1, r, u << 1 | 1));
        return res;
    }
}


int main() {
    cin >> n >> q;
    
    for (int i = 1; i <= n; i ++ )
        scanf("%d", &arr[i]);
    
    build(1, n, 1);
    
    char op[2];
    int a, b;
    while (q -- ) {
        scanf("%s%d%d", op, &a, &b);
        if (op[0] == 'Q') {
            printf("%lld\n", query(a, b, 1, n, 1));
        } else {
            if (arr[a] < b) {
                modify(a, a, b, 1, n, 1);
                arr[a] = b;
            }
        }
    }
    
    return 0;
}
```
