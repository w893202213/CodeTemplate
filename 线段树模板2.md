#### 线段树模板2：单点修改(加值)、区间查询(求最大值)
#### 不带懒标记

代码：

```cpp
#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

typedef long long LL;

const int N = 2e5 + 10;

LL arr[N];
LL tree[N << 2];

void up(int u) {
    tree[u] = max(tree[u << 1], tree[u << 1 | 1]);
}

void build(int l, int r, int u) {
    if (l == r)
        tree[u] = arr[l];
    else {
        int mid = l + r >> 1;
        
        build(l, mid, u << 1);
        build(mid + 1, r, u << 1 | 1);
        
        up(u);
    }
}

// 将下标 pos 处的值加上 jobv (最大值当然也加上了 jobv )
void add(int pos, LL jobv, int l, int r, int u) {
    if (l == r)
        tree[u] += jobv;
    else {
        int mid = l + r >> 1;
        
        if (pos <= mid)
            add(pos, jobv, l, mid, u << 1);
        else
            add(pos, jobv, mid + 1, r, u << 1 | 1);
        
        up(u);
    }
}

LL query(int jobl, int jobr, int l, int r, int u) {
    if (jobl <= l && r <= jobr)
        return tree[u];
    else {
        int mid = l + r >> 1;
        
        LL res = -2e9;
        if (jobl <= mid)
            res = max(res, query(jobl, jobr, l, mid, u << 1));
        if (mid + 1 <= jobr)
            res = max(res, query(jobl, jobr, mid + 1, r, u << 1 | 1));
        
        return res;
    }
}

int main() {
    int m, d;
    LL t = 0;
    LL len = 0;  // 当前数组长度
    int n = 2e5 + 5;
    
    cin >> m >> d;
    
    build(1, n, 1);  // 先将空格摆好
    
    char op[2];
    LL x;
    
    while (m -- ) {
        scanf("%s%lld", op, &x);
        if (*op == 'Q') {
            t = query(len - x + 1, len, 1, n, 1);
            printf("%lld\n", t);
        } else {
            x = ((x + t) % d + d) % d;
            len ++;
            add(len, x, 1, n, 1);
        }
    }
    
    return 0;
}
```
