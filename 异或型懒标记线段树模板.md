#### 数组元素为 bool 值，初始全为 false
#### 支持区间反转+区间求和操作

代码：

```cpp
#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

typedef long long LL;

const int N = 1e5 + 10;

int n, q;
LL tree[N << 2];  // 维护区间和
LL st[N << 2];  // 当前区间是否需要反转

void up(int u) {
    tree[u] = tree[u << 1] + tree[u << 1 | 1];
}

void lazy(int u, int cnt) {
    tree[u] = cnt - tree[u];
    st[u] = !st[u];  // !!! 由于 "反转" 操作可能有覆盖, 因此这里只能是严格的 st[u] = !st[u];  
}

void down(int u, int lcnt, int rcnt) {
    if (st[u]) {
        lazy(u << 1, lcnt);
        lazy(u << 1 | 1, rcnt);
        st[u] = false;
    }
}

void build(int l, int r, int u) {
    if (l == r)
        tree[u] = 0;
    else {
        int mid = l + r >> 1;
        
        build(l, mid, u << 1);
        build(mid + 1, r, u << 1 | 1);
        
        up (u);
    }
    
    st[u] = false;
}

// 反转区间 [jobl, jobr] 内的值
void modify(int jobl, int jobr, int l, int r, int u) {
    if (jobl <= l && r <= jobr)
        lazy(u, r - l + 1);
    else {
        int mid = l + r >> 1;
        
        down(u, mid - l + 1, r - mid);
        
        if (jobl <= mid)
            modify(jobl, jobr, l, mid, u << 1);
        if (mid + 1 <= jobr)
            modify(jobl, jobr, mid + 1, r, u << 1 | 1);
        
        up(u);
    }
}

LL query(int jobl, int jobr, int l, int r, int u) {
    if (jobl <= l && r <= jobr)
        return tree[u];
    else {
        int mid = l + r >> 1;
        
        down(u, mid - l + 1, r - mid);
        
        LL res = 0;
        if (jobl <= mid)
            res += query(jobl, jobr, l, mid, u << 1);
        if (mid + 1 <= jobr)
            res += query(jobl, jobr, mid + 1, r, u << 1 | 1);
        
        return res;
    }
}

int main() {
    cin >> n >> q;
    
    build(1, n, 1);
    
    int op, l, r;
    while (q -- ) {
        scanf("%d%d%d", &op, &l, &r);
        if (op == 0) {
            modify(l, r, 1, n, 1);
        } else {
            printf("%lld\n", query(l, r, 1, n, 1));
        }
    }
    
    return 0;
}
```
