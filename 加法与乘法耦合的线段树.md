### 本题是洛谷模板题2：P3373

### 两个关键点：
    1. 由于加法操作和乘法操作会相互耦合，因此需要维护统一的 down 逻辑
    2. 所有的中间的加法、乘法计算都要取模，防止溢出



代码：

```cpp
#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

typedef long long LL;

const int N = 1e5 + 10;
int MOD;

int n, q;

LL arr[N];
LL tree[N << 2];  // 维护区间和
LL multi_val[N << 2];
LL add_val[N << 2];

void up(int u) {
    tree[u] = tree[u << 1] + tree[u << 1 | 1];
}

// 维护统一的 down 逻辑
void down(int u, int l, int r) {
    int mid = l + r >> 1;
    
    // 下传给左儿子
    // 左儿子新和 = (左原和 * 父乘) + (父加 * 左长度) 
    tree[u << 1] = (tree[u << 1] * multi_val[u] + add_val[u] * (mid - l + 1)) % MOD;
    multi_val[u << 1] = (multi_val[u << 1] * multi_val[u]) % MOD;
    add_val[u << 1] = (add_val[u << 1] * multi_val[u] + add_val[u]) % MOD;
    
    // 下传给右儿子
    // 同样的逻辑
    tree[u << 1 | 1] = (tree[u << 1 | 1] * multi_val[u] + add_val[u] * (r - mid)) % MOD;
    multi_val[u << 1 | 1] = (multi_val[u << 1 | 1] * multi_val[u]) % MOD;
    add_val[u << 1 | 1] = (add_val[u << 1 | 1] * multi_val[u] + add_val[u]) % MOD;
    
    // 重置父节点标记
    multi_val[u] = 1;
    add_val[u] = 0;
}

void lazy_multi(int u, LL val) {
    tree[u] = (tree[u] * val) % MOD;
    multi_val[u] = (multi_val[u] * val) % MOD;
    add_val[u] = (add_val[u] * val) % MOD;  // 关键!!! 加法标记也要乘 val 
}


void lazy_add(int u, LL val, int len) {
    tree[u] = (tree[u] + val * len) % MOD;
    add_val[u] = (add_val[u] + val) % MOD;
}

void build(int l, int r, int u) {
    if (l == r)
        tree[u] = arr[l];
    else {
        int mid = l + r >> 1;
        
        build(l, mid, u << 1);
        build(mid + 1, r, u << 1 | 1);
        
        up(u);
    }
    
    multi_val[u] = 1;
    add_val[u] = 0;
}

void multi(int jobl, int jobr, LL jobv, int l, int r, int u) {
    if (jobl <= l && r <= jobr)
        lazy_multi(u, jobv);
    else {
        int mid = l + r >> 1;
        
        down(u, l, r);
        
        if (jobl <= mid)
            multi(jobl, jobr, jobv, l, mid, u << 1);
        if (mid + 1 <= jobr)
            multi(jobl, jobr, jobv, mid + 1, r, u << 1 | 1);
        
        up(u);
    }
}

void add(int jobl, int jobr, LL jobv, int l, int r, int u) {
    if (jobl <= l && r <= jobr)
        lazy_add(u, jobv, r - l + 1);
    else {
        int mid = l + r >> 1;
        
        down(u, l, r);
        
        if (jobl <= mid)
            add(jobl, jobr, jobv, l, mid, u << 1);
        if (mid + 1 <= jobr)
            add(jobl, jobr, jobv, mid + 1, r, u << 1 | 1);
        
        up(u);
    }
}

LL query(int jobl, int jobr, int l, int r, int u) {
    if (jobl <= l && r <= jobr)
        return tree[u];
    else {
        int mid = l + r >> 1;
        
        down(u, l, r);
        
        LL res = 0;
        if (jobl <= mid)
            res = (res + query(jobl, jobr, l, mid, u << 1)) % MOD;
        if (mid + 1 <= jobr)
            res = (res + query(jobl, jobr, mid + 1, r, u << 1 | 1)) % MOD;
        
        return res;
    }
}

int main() {
    cin >> n >> q >> MOD;
    
    for (int i = 1; i <= n; i ++ )
        scanf("%lld", &arr[i]);
    
    build(1, n, 1);
    
    int op, x, y;
    LL k;
    while (q -- ) {
        scanf("%d", &op);
        if (op == 1) {
            scanf("%d%d%lld", &x, &y, &k);
            multi(x, y, k, 1, n, 1);
        } else if (op == 2) {
            scanf("%d%d%lld", &x, &y, &k);
            add(x, y, k, 1, n, 1);
        } else {
            scanf("%d%d", &x, &y);
            printf("%lld\n", query(x, y, 1, n, 1) % MOD);
        }
    }
    
    return 0;
}
```
