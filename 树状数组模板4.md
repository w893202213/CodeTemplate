#### 典型离线树状数组: 洛谷P1792
#### 实现多次询问区间 [l, r] 中不同数字的个数。

做法：

将询问按右端点 r 从小到大排序；

从左到右扫描原数组；

对于当前数字 a[i]：

如果它之前出现过，就把上一次出现位置的贡献清掉；再把当前位置贡献设为 1；

对于询问 [l, r]，答案就是：query(r) - query(l - 1)

代码：

```cpp
#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
#include <string>
#include <unordered_map>

using namespace std;

typedef long long LL;

const int N = 1e6 + 10;

int n;

int colors[N];  // 记录每个位置的颜色种类

struct Ask {
    int l, r;
    int index;
};

Ask queries[N];

int res[N];

LL tr[N];

int lowbit(int x) {
    return x & -x;
}

void add(int x, LL c) {
    for (int i = x; i <= n; i += lowbit(i))
        tr[i] += c;
}

LL query(int x) {
    LL res = 0;
    for (int i = x; i; i -= lowbit(i))
        res += tr[i];
    return res;
}

LL range_query(int l, int r) {
    return query(r) - query(l - 1);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    
    cin >> n;
    
    for (int i = 1; i <= n; i ++ )
        cin >> colors[i];
    
    int m;
    cin >> m;
    
    // 读入并存下所有的询问
    int l, r, index;
    for (int i = 1; i <= m; i ++ ) {
        cin >> l >> r;
        queries[i] = {l, r, i};
    }
    
    // 将所有询问按照右端点排序
    sort(queries + 1, queries + m + 1, [&] (const Ask &a, const Ask &b) {
        return a.r < b.r;
    });
    
    // 同时遍历原数组和询问数组，并更新答案
    unordered_map<int, int> last;
    for (int s = 1, t = 1; t <= m; t ++ ) {
        l = queries[t].l, r = queries[t].r, index = queries[t].index;
        
        for (; s <= r; s ++ ) {
            int color = colors[s];
            if (last.count(color)) {
                add(last[color], -1);
            }
            add(s, 1);
            last[color] = s;
        }
        
        res[index] = range_query(l, r);
    }
    
    for (int i = 1; i <= m; i ++ )
        cout << res[i] << "\n";
    
    return 0;
}
```
